229  }
230  //printf("%s::m_allMembersDict->append(%s)\n",name().data(),md->localName().data());
231  m_allMembersDict->append(md->localName(),md);
232  //::addNamespaceMemberNameToIndex(md);
233  //static bool sortBriefDocs=Config_getBool(SORT_BRIEF_DOCS);
234  switch(md->memberType())
235  {
236  case MemberType_Variable:
237  addMemberToList(MemberListType_decVarMembers,md);
238  addMemberToList(MemberListType_docVarMembers,md);
239  break;
240  case MemberType_Function:
241  addMemberToList(MemberListType_decFuncMembers,md);
242  addMemberToList(MemberListType_docFuncMembers,md);
243  break;
244  case MemberType_Typedef:
245  addMemberToList(MemberListType_decTypedefMembers,md);
246  addMemberToList(MemberListType_docTypedefMembers,md);
247  break;
248  case MemberType_Enumeration:
249  addMemberToList(MemberListType_decEnumMembers,md);
250  addMemberToList(MemberListType_docEnumMembers,md);
251  break;
252  case MemberType_EnumValue:
253  break;
254  case MemberType_Define:
255  addMemberToList(MemberListType_decDefineMembers,md);
256  addMemberToList(MemberListType_docDefineMembers,md);
257  break;
258  default:
259  err("NamespaceDef::insertMembers(): "
260  "member `%s' with class scope `%s' inserted in namespace scope `%s'!\n",

310  }
311  }

314  {
315  if (classSDict)

317  SDict<ClassDef>::Iterator ci(*classSDict);
318  ClassDef *cd;
319  for (ci.toFirst();(cd=ci.current());++ci)
320  {
321  if (cd->isLinkableInProject())
322  {
323  tagFile << " <class kind=\"" << cd->compoundTypeString()
324  << "\">" << convertToXML(cd->name()) << "</class>" << endl;
325  }
326  }
327  }
328  }
329  case LayoutDocEntry::MemberDecl:
330  {
331  LayoutDocEntryMemberDecl *lmd = (LayoutDocEntryMemberDecl*)lde;
332  MemberList * ml = getMemberList(lmd->type);
333  if (ml)
334  {
335  ml->writeTagFile(tagFile);
336  }
337  }
338  break;
339  case LayoutDocEntry::MemberGroups:
340  {
341  if (memberGroupSDict)
342  {
343  MemberGroupSDict::Iterator mgli(*memberGroupSDict);
344  MemberGroup *mg;
345  for (;(mg=mgli.current());++mgli)
346  {
347  mg->writeTagFile(tagFile);
348  }
349  }
350  }
351  break;

353  break;
354  }
355  }

569 void NamespaceDef::writeDocumentation(OutputList &ol)
570 {
571  static bool generateTreeView = Config_getBool(GENERATE_TREEVIEW);
572  //static bool outputJava = Config_getBool(OPTIMIZE_OUTPUT_JAVA);
573  //static bool fortranOpt = Config_getBool(OPTIMIZE_FOR_FORTRAN);
574 
575  QCString pageTitle = title();
576  startFile(ol,getOutputFileBase(),name(),pageTitle,HLI_NamespaceVisible,!generateTreeView);
577 
578  if (!generateTreeView)
579  {
580  if (getOuterScope()!=Doxygen::globalScope)
581  {
582  writeNavigationPath(ol);
583  }
584  ol.endQuickIndices();
585  }
586 
587  startTitle(ol,getOutputFileBase(),this);
588  ol.parseText(pageTitle);
589  addGroupListToTitle(ol,this);
590  addNamespaceAttributes(ol);
591  endTitle(ol,getOutputFileBase(),displayName());
592  ol.startContents();
593 
594  if (Doxygen::searchIndex)
595  {
596  Doxygen::searchIndex->setCurrentDoc(this,anchor(),FALSE);
597  Doxygen::searchIndex->addWord(localName(),TRUE);
598  }
599 
600  Doxygen::indexList->addIndexItem(this,0);
601 
602  //---------------------------------------- start flexible part -------------------------------
603 
604  SrcLangExt lang = getLanguage();
605  QListIterator<LayoutDocEntry> eli(
606  LayoutDocManager::instance().docEntries(LayoutDocManager::Namespace));
607  LayoutDocEntry *lde;
608  for (eli.toFirst();(lde=eli.current());++eli)
609  {
610  switch (lde->kind())
611  {
612  case LayoutDocEntry::BriefDesc:
613  writeBriefDescription(ol);
614  break;
615  case LayoutDocEntry::MemberDeclStart:
616  startMemberDeclarations(ol);
617  break;
618  case LayoutDocEntry::NamespaceClasses:
619  {
620  LayoutDocEntrySection *ls = (LayoutDocEntrySection*)lde;
621  writeClassDeclarations(ol,ls->title(lang));
622  }
623  break;
624  case LayoutDocEntry::NamespaceNestedNamespaces:
625  {
626  LayoutDocEntrySection *ls = (LayoutDocEntrySection*)lde;
627  writeNamespaceDeclarations(ol,ls->title(lang),false);
628  }
629  break;
630  case LayoutDocEntry::NamespaceNestedConstantGroups:
631  {
632  LayoutDocEntrySection *ls = (LayoutDocEntrySection*)lde;
633  writeNamespaceDeclarations(ol,ls->title(lang),true);
634  }
635  break;
636  case LayoutDocEntry::MemberGroups:
637  writeMemberGroups(ol);
638  break;
639  case LayoutDocEntry::MemberDecl:
640  {
641  LayoutDocEntryMemberDecl *lmd = (LayoutDocEntryMemberDecl*)lde;
642  writeMemberDeclarations(ol,lmd->type,lmd->title(lang));
643  }
644  break;
645  case LayoutDocEntry::MemberDeclEnd:
646  endMemberDeclarations(ol);
647  break;
648  case LayoutDocEntry::DetailedDesc:
649  {
650  LayoutDocEntrySection *ls = (LayoutDocEntrySection*)lde;
651  writeDetailedDescription(ol,ls->title(lang));
652  }
653  break;
654  case LayoutDocEntry::MemberDefStart:
655  startMemberDocumentation(ol);
656  break;
657  case LayoutDocEntry::NamespaceInlineClasses:
658  writeInlineClasses(ol);
659  break;
660  case LayoutDocEntry::MemberDef:
661  {
662  LayoutDocEntryMemberDef *lmd = (LayoutDocEntryMemberDef*)lde;
663  writeMemberDocumentation(ol,lmd->type,lmd->title(lang));
664  }
665  break;
666  case LayoutDocEntry::MemberDefEnd:
667  endMemberDocumentation(ol);
668  break;
669  case LayoutDocEntry::AuthorSection:
670  writeAuthorSection(ol);
671  break;
672  case LayoutDocEntry::ClassIncludes:
673  case LayoutDocEntry::ClassInheritanceGraph:
674  case LayoutDocEntry::ClassNestedClasses:
675  case LayoutDocEntry::ClassCollaborationGraph:
676  case LayoutDocEntry::ClassAllMembersLink:
677  case LayoutDocEntry::ClassUsedFiles:
678  case LayoutDocEntry::ClassInlineClasses:
679  case LayoutDocEntry::FileClasses:
680  case LayoutDocEntry::FileNamespaces:
681  case LayoutDocEntry::FileConstantGroups:
682  case LayoutDocEntry::FileIncludes:
683  case LayoutDocEntry::FileIncludeGraph:
684  case LayoutDocEntry::FileIncludedByGraph:
685  case LayoutDocEntry::FileSourceLink:
686  case LayoutDocEntry::FileInlineClasses:
687  case LayoutDocEntry::GroupClasses:
688  case LayoutDocEntry::GroupInlineClasses:
689  case LayoutDocEntry::GroupNamespaces:
690  case LayoutDocEntry::GroupDirs:
691  case LayoutDocEntry::GroupNestedGroups:
692  case LayoutDocEntry::GroupFiles:
693  case LayoutDocEntry::GroupGraph:
694  case LayoutDocEntry::GroupPageDocs:
695  case LayoutDocEntry::DirSubDirs:
696  case LayoutDocEntry::DirFiles:
697  case LayoutDocEntry::DirGraph:
698  err("Internal inconsistency: member %d should not be part of "
699  "LayoutDocManager::Namespace entry list\n",lde->kind());
700  break;
701  }
702  }
703 
704  //---------------------------------------- end flexible part -------------------------------
705 
706  ol.endContents();
707 
708  endFileWithNavPath(this,ol);
709 
710  if (Config_getBool(SEPARATE_MEMBER_PAGES))
711  {
712  MemberList *allMemberList = getMemberList(MemberListType_allMembersList);
713  if (allMemberList) allMemberList->sort();
714  writeMemberPages(ol);
715  }
716 }
