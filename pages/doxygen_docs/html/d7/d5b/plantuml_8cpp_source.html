27 QCString writePlantUMLSource(const QCString &outDir,const QCString &fileName,const QCString &content)
28 {
29  QCString baseName(4096);
30  static int umlindex=1;
31 
32  if (fileName.isEmpty()) // generate name
33  {
34  baseName = outDir+"/inline_umlgraph_"+QCString().setNum(umlindex++);
35  }
36  else // user specified name
37  {
38  baseName = fileName;
39  int i=baseName.findRev('.');
40  if (i!=-1) baseName = baseName.left(i);
41  baseName.prepend(outDir+"/");
42  }
43  QFile file(baseName+".pu");
44  if (!file.open(IO_WriteOnly))
45  {
46  err("Could not open file %s for writing\n",baseName.data());
47  }
48  QCString text = "@startuml\n";
49  text+=content;
50  text+="@enduml\n";
51  file.writeBlock( text, text.length() );
52  file.close();
53  return baseName;
54 }

56 void generatePlantUMLOutput(const char *baseName,const char *outDir,PlantUMLOutputFormat format)
57 {
58  static QCString plantumlJarPath = Config_getString(PLANTUML_JAR_PATH);
59  static QCString plantumlConfigFile = Config_getString(PLANTUML_CFG_FILE);
60 
61  QCString pumlExe = "java";
62  QCString pumlArgs = "";
63 
64  QStrList &pumlIncludePathList = Config_getList(PLANTUML_INCLUDE_PATH);
65  char *s=pumlIncludePathList.first();
66  if (s)
67  {
68  pumlArgs += "-Dplantuml.include.path=\"";
69  pumlArgs += s;
70  s = pumlIncludePathList.next();
71  }
72  while (s)
73  {
74  pumlArgs += portable_pathListSeparator();
75  pumlArgs += s;
76  s = pumlIncludePathList.next();
77  }
78  if (pumlIncludePathList.first()) pumlArgs += "\" ";
79  pumlArgs += "-Djava.awt.headless=true -jar \""+plantumlJarPath+"plantuml.jar\" ";
80  if (!plantumlConfigFile.isEmpty())
81  {
82  pumlArgs += "-config \"";
83  pumlArgs += plantumlConfigFile;
84  pumlArgs += "\" ";
85  }
86  pumlArgs+="-o \"";
87  pumlArgs+=outDir;
88  pumlArgs+="\" ";
89  QCString imgName = baseName;
90  // The basename contains path, we need to strip the path from the filename in order
91  // to create the image file name which should be included in the index.qhp (Qt help index file).
92  int i;
93  if ((i=imgName.findRev('/'))!=-1) // strip path
94  {
95  imgName=imgName.right(imgName.length()-i-1);
96  }
97  switch (format)
98  {
99  case PUML_BITMAP:
100  pumlArgs+="-tpng";
101  imgName+=".png";
102  break;
103  case PUML_EPS:
104  pumlArgs+="-teps";
105  imgName+=".eps";
106  break;
107  case PUML_SVG:
108  pumlArgs+="-tsvg";
109  imgName+=".svg";
110  break;
111  }
112  pumlArgs+=" \"";
113  pumlArgs+=baseName;
114  pumlArgs+=".pu\" ";
115  pumlArgs+="-charset UTF-8 ";
116  int exitCode;
117  //printf("*** running: %s %s outDir:%s %s\n",pumlExe.data(),pumlArgs.data(),outDir,baseName);
118  msg("Running PlantUML on generated file %s.pu\n",baseName);
119  portable_sysTimerStart();
120  if ((exitCode=portable_system(pumlExe,pumlArgs,TRUE))!=0)
121  {
122  err("Problems running PlantUML. Verify that the command 'java -jar \"%splantuml.jar\" -h' works from the command line. Exit code: %d\n",
123  plantumlJarPath.data(),exitCode);
124  }
125  else if (Config_getBool(DOT_CLEANUP))
126  {
127  QFile(QCString(baseName)+".pu").remove();
128  }
129  portable_sysTimerStop();
130  if ( (format==PUML_EPS) && (Config_getBool(USE_PDFLATEX)) )
131  {
132  QCString epstopdfArgs(maxCmdLine);
133  epstopdfArgs.sprintf("\"%s.eps\" --outfile=\"%s.pdf\"",baseName,baseName);
134  portable_sysTimerStart();
135  if ((exitCode=portable_system("epstopdf",epstopdfArgs))!=0)
136  {
137  err("Problems running epstopdf. Check your TeX installation! Exit code: %d\n",exitCode);
138  }
139  portable_sysTimerStop();
140  }
141  Doxygen::indexList->addImageFile(imgName);
142 }
