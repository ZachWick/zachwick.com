95 {
96  bool found = name()=="citelist";

98  RefList *rl;
99  for (rli.toFirst();(rl=rli.current()) && !found;++rli)

102  {
103  found=TRUE;
104  break;
105  }
106  }
107  if (!found) // not one of the generated related pages
108  {
109  tagFile << " <compound kind=\"page\">" << endl;

113  writeDocAnchorsToTagFile(tagFile);
114  tagFile << " </compound>" << endl;
115  }

118 void PageDef::writeDocumentation(OutputList &ol)
119 {
120  static bool generateTreeView = Config_getBool(GENERATE_TREEVIEW);
121 
122  //outputList->disable(OutputGenerator::Man);
123  QCString pageName,manPageName;
124  pageName = escapeCharsInString(name(),FALSE,TRUE);
125  manPageName = escapeCharsInString(name(),TRUE,TRUE);
126 
127  //printf("PageDef::writeDocumentation: %s\n",getOutputFileBase().data());
128 
129  ol.pushGeneratorState();
130  //1.{
131 
132  if (m_nestingLevel>0
133  //&& // a sub page
134  //(Doxygen::mainPage==0 || getOuterScope()!=Doxygen::mainPage) // and not a subpage of the mainpage
135  )
136  {
137  // do not generate sub page output for RTF and LaTeX, as these are
138  // part of their parent page
139  ol.disableAll();
140  ol.enable(OutputGenerator::Man);
141  ol.enable(OutputGenerator::Html);
142  }
143 
144  ol.pushGeneratorState();
145  //2.{
146  ol.disableAllBut(OutputGenerator::Man);
147  startFile(ol,getOutputFileBase(),manPageName,title(),HLI_Pages,!generateTreeView);
148  ol.enableAll();
149  ol.disable(OutputGenerator::Man);
150  startFile(ol,getOutputFileBase(),pageName,title(),HLI_Pages,!generateTreeView);
151  ol.popGeneratorState();
152  //2.}
153 
154  if (!generateTreeView)
155  {
156  if (getOuterScope()!=Doxygen::globalScope && !Config_getBool(DISABLE_INDEX))
157  {
158  getOuterScope()->writeNavigationPath(ol);
159  }
160  ol.endQuickIndices();
161  }
162  SectionInfo *si=Doxygen::sectionDict->find(name());
163 
164  // save old generator state and write title only to Man generator
165  ol.pushGeneratorState();
166  //2.{
167  ol.disableAllBut(OutputGenerator::Man);
168  ol.startTitleHead(manPageName);
169  ol.endTitleHead(manPageName, manPageName);
170  if (si)
171  {
172  ol.generateDoc(docFile(),docLine(),this,0,si->title,TRUE,FALSE,0,TRUE,FALSE);
173  ol.endSection(si->label,si->type);
174  }
175  ol.popGeneratorState();
176  //2.}
177 
178  // for Latex the section is already generated as a chapter in the index!
179  ol.pushGeneratorState();
180  //2.{
181  ol.disable(OutputGenerator::Latex);
182  ol.disable(OutputGenerator::RTF);
183  ol.disable(OutputGenerator::Man);
184  if (!title().isEmpty() && !name().isEmpty() && si!=0)
185  {
186  //ol.startSection(si->label,si->title,si->type);
187  startTitle(ol,getOutputFileBase(),this);
188  ol.generateDoc(docFile(),docLine(),this,0,si->title,TRUE,FALSE,0,TRUE,FALSE);
189  //stringToSearchIndex(getOutputFileBase(),
190  // theTranslator->trPage(TRUE,TRUE)+" "+si->title,
191  // si->title);
192  //ol.endSection(si->label,si->type);
193  endTitle(ol,getOutputFileBase(),name());
194  }
195  ol.startContents();
196  ol.popGeneratorState();
197  //2.}
198 
199  if (m_showToc && hasSections())
200  {
201  writeToc(ol);
202  }
203 
204  writePageDocumentation(ol);
205 
206  if (generateTreeView && getOuterScope()!=Doxygen::globalScope && !Config_getBool(DISABLE_INDEX))
207  {
208  ol.endContents();
209  endFileWithNavPath(getOuterScope(),ol);
210  }
211  else
212  {
213  endFile(ol);
214  }
215 
216  ol.popGeneratorState();
217  //1.}
218 
219  Doxygen::indexList->addIndexItem(this,0,0,filterTitle(title()));
220 }

222 void PageDef::writePageDocumentation(OutputList &ol)
223 {
224 
225  bool markdownEnabled = Doxygen::markdownSupport;
226  if (getLanguage()==SrcLangExt_Markdown)
227  {
228  Doxygen::markdownSupport = TRUE;
229  }
230 
231  ol.startTextBlock();
232  QCString docStr = documentation()+inbodyDocumentation();
233  if (!docStr.isEmpty())
234  {
235  ol.pushGeneratorState();
236  ol.disableAllBut(OutputGenerator::Man);
237  ol.writeString(" - ");
238  ol.popGeneratorState();
239  }
240  ol.generateDoc(
241  docFile(), // fileName
242  docLine(), // startLine
243  this, // context
244  0, // memberdef
245  docStr, // docStr
246  TRUE, // index words
247  FALSE // not an example
248  );
249  ol.endTextBlock();
250 
251  Doxygen::markdownSupport = markdownEnabled;
252 
253  if (hasSubPages())
254  {
255  // for printed documentation we write subpages as section's of the
256  // parent page.
257  ol.pushGeneratorState();
258  ol.disableAll();
259  ol.enable(OutputGenerator::Latex);
260  ol.enable(OutputGenerator::RTF);
261 
262  PageSDict::Iterator pdi(*m_subPageDict);
263  PageDef *subPage=pdi.toFirst();
264  for (pdi.toFirst();(subPage=pdi.current());++pdi)
265  {
266  SectionInfo::SectionType sectionType = SectionInfo::Paragraph;
267  switch (m_nestingLevel)
268  {
269  case 0: sectionType = SectionInfo::Page; break;
270  case 1: sectionType = SectionInfo::Section; break;
271  case 2: sectionType = SectionInfo::Subsection; break;
272  case 3: sectionType = SectionInfo::Subsubsection; break;
273  default: sectionType = SectionInfo::Paragraph; break;
274  }
275  QCString title = subPage->title();
276  if (title.isEmpty()) title = subPage->name();
277  ol.startSection(subPage->name(),title,sectionType);
278  ol.parseText(title);
279  ol.endSection(subPage->name(),sectionType);
280  Doxygen::subpageNestingLevel++;
281  subPage->writePageDocumentation(ol);
282  Doxygen::subpageNestingLevel--;
283  }
284 
285  ol.popGeneratorState();
286  }
287 }
