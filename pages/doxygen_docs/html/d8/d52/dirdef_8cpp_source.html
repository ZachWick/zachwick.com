843 static void computeCommonDirPrefix()
844 {
845  QCString path;
846  DirDef *dir;
847  DirSDict::Iterator sdi(*Doxygen::directories);
848  if (Doxygen::directories->count()>0) // we have at least one dir
849  {
850  // start will full path of first dir
851  sdi.toFirst();
852  dir=sdi.current();
853  path=dir->name();
854  int i=path.findRev('/',path.length()-2);
855  path=path.left(i+1);
856  bool done=FALSE;
857  if (i==-1)
858  {
859  path="";
860  }
861  else
862  {
863  while (!done)
864  {
865  int l = path.length();
866  int count=0;
867  for (sdi.toFirst();(dir=sdi.current());++sdi)
868  {
869  QCString dirName = dir->name();
870  if (dirName.length()>path.length())
871  {
872  if (qstrncmp(dirName,path,l)!=0) // dirName does not start with path
873  {
874  int i=path.findRev('/',l-2);
875  if (i==-1) // no unique prefix -> stop
876  {
877  path="";
878  done=TRUE;
879  }
880  else // restart with shorter path
881  {
882  path=path.left(i+1);
883  break;
884  }
885  }
886  }
887  else // dir is shorter than path -> take path of dir as new start
888  {
889  path=dir->name();
890  l=path.length();
891  int i=path.findRev('/',l-2);
892  if (i==-1) // no unique prefix -> stop
893  {
894  path="";
895  done=TRUE;
896  }
897  else // restart with shorter path
898  {
899  path=path.left(i+1);
900  }
901  break;
902  }
903  count++;
904  }
905  if (count==Doxygen::directories->count())
906  // path matches for all directories -> found the common prefix
907  {
908  done=TRUE;
909  }
910  }
911  }
912  }
913  for (sdi.toFirst();(dir=sdi.current());++sdi)
914  {
915  QCString diskName = dir->name().right(dir->name().length()-path.length());
916  dir->setDiskName(diskName);
917  //printf("set disk name: %s -> %s\n",dir->name().data(),diskName.data());
918  }
919 }
